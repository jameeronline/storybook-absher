export const ColorSwatch = ({ colors }) => {
  return (
    <article className="color-section flex flex-wrap gap-4">
      {colors.map((item) => {
        return Object.entries(item).map(([key, value]) => (
          <ColorBlock colorHex={value} colorHsl={value} name={key} key={key} />
        ));
      })}
    </article>
  );
};


export const ColorBlock = ({ colorHex, colorHsl, name }) => {
  const copyToClipboard = (value) => {
    navigator.clipboard.writeText(value);
    alert(`Copied ${value} to clipboard!`);
  };

  return (
    <article className='flex justify-center flex-col items-center gap-4'>
      <div
        className="color-block"
        style={{ backgroundColor: colorHex }}
        onClick={() => copyToClipboard(colorHex)}></div>
      <span className="flex flex-col gap-1 items-center text-xs">
        <span className='font-bold'>{name}</span>
        <span className='inline-flex gap-1'>
          {colorHex}
          <button
            onClick={(e) => {
              e.stopPropagation(); // Prevent double copying
              console.log('copied');
              toast(`copied ${colorHex} to clipboard`)
              //copyToClipboard(colorHex);
            }}
          >
            <i className="bi bi-copy"></i>
          </button>
        </span>
      </span>
    </article>
  );
};
